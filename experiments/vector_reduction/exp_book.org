#+begin_src R :tangle yes :session :results none
library(DoE.base)
library(tidyverse)

size = c("P", "M", "G")
nb = c("P", "M", "G")
fr = c("P", "M", "G")

fac.design (
    nfactors=3,
    replications=10,
    repeat.only=FALSE,
    blocks=1,
    randomize=TRUE,
    seed=10373,
    factor.names=list(
        Size=size,
        NumberOfBlocks=nb,
        ReductionFactor=fr)) %>%
  as_tibble %>%
  select(-Blocks) %>%
  write_csv("exp01/runs.csv")
